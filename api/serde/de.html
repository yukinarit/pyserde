 102def serde_legacy_custom_class_deserializer(
 103    cls: type[Any], datavar: Any, value: Any, custom: DeserializeFunc, default: Callable[[], Any]
 104) -> Any:
 105    """
 106    Handle custom deserialization. Use default deserialization logic if it receives `SerdeSkip`
 107    exception.
 108
 109    :param cls: Type of the field.
 110    :param datavar: The whole variable to deserialize from. e.g. "data"
 111    :param value: The value for the field. e.g. "data['i']"
 112    :param custom: Custom deserialize function.
 113    :param default: Default deserialize function.
 114    """
 115    try:
 116        return custom(cls, value)
 117    except SerdeSkip:
 118        return default()

 129def _get_by_aliases(
 130    d: dict[str, str], aliases: list[str], raise_error: bool = True
 131) -> Optional[str]:
 132    if not aliases:
 133        if raise_error:
 134            raise KeyError("Tried all aliases, but key not found")
 135        else:
 136            return None
 137    if aliases[0] in d:
 138        return d[aliases[0]]
 139    else:
 140        return _get_by_aliases(d, aliases[1:], raise_error=raise_error)
 141
 142
 143def _exists_by_aliases(d: dict[str, str], aliases: list[str]) -> bool:
 144    for alias in aliases:
 145        if alias in d:
 146            return True
 147    return False

 150def _make_deserialize(
 151    cls_name: str,
 152    fields: list[Any],
 153    *args: Any,
 154    rename_all: Optional[str] = None,
 155    reuse_instances_default: bool = True,
 156    convert_sets_default: bool = False,
 157    deserializer: Optional[DeserializeFunc] = None,
 158    type_check: TypeCheck = strict,
 159    class_deserializer: Optional[ClassDeserializer] = None,
 160    **kwargs: Any,
 161) -> type[Any]:
 162    """
 163    Create a deserializable class programatically.
 164    """
 165    C = dataclasses.make_dataclass(cls_name, fields, *args, **kwargs)
 166    C = deserialize(
 167        C,
 168        rename_all=rename_all,
 169        reuse_instances_default=reuse_instances_default,
 170        convert_sets_default=convert_sets_default,
 171        **kwargs,
 172    )
 173    return C

 176# The `deserialize` function can call itself recursively when it needs to generate code for
 177# unmarked dataclasses. To avoid infinite recursion, this array remembers types for which code is
 178# currently being generated.
 179GENERATION_STACK = []

 182@dataclass_transform()
 183def deserialize(
 184    _cls: Optional[type[T]] = None,
 185    rename_all: Optional[str] = None,
 186    reuse_instances_default: bool = True,
 187    convert_sets_default: bool = False,
 188    deserializer: Optional[DeserializeFunc] = None,
 189    tagging: Tagging = DefaultTagging,
 190    type_check: TypeCheck = strict,
 191    class_deserializer: Optional[ClassDeserializer] = None,
 192    **kwargs: Any,
 193) -> type[T]:
 194    """
 195    A dataclass with this decorator is deserializable from any of the data formats supported
 196    by pyserde.
 197
 198    >>> from serde import deserialize
 199    >>> from serde.json import from_json
 200    >>>
 201    >>> @deserialize
 202    ... class Foo:
 203    ...     i: int
 204    ...     s: str
 205    ...     f: float
 206    ...     b: bool
 207    >>>
 208    >>> from_json(Foo, '{"i": 10, "s": "foo", "f": 100.0, "b": true}')
 209    Foo(i=10, s='foo', f=100.0, b=True)
 210    """
 211
 212    stack = []
 213
 214    def wrap(cls: type[T]) -> type[T]:
 215        if cls in stack:
 216            return cls
 217        stack.append(cls)
 218
 219        tagging.check()
 220
 221        # If no `dataclass` found in the class, dataclassify it automatically.
 222        if not is_dataclass(cls):
 223            dataclass(cls)
 224
 225        if type_check.is_strict():
 226            beartype.beartype(cls)
 227
 228        g: dict[str, Any] = {}
 229
 230        # Create a scope storage used by serde.
 231        # Each class should get own scope. Child classes can not share scope with parent class.
 232        # That's why we need the "scope.cls is not cls" check.
 233        scope: Optional[Scope] = getattr(cls, SERDE_SCOPE, None)
 234        if scope is None or scope.cls is not cls:
 235            scope = Scope(cls, reuse_instances_default=reuse_instances_default)
 236            setattr(cls, SERDE_SCOPE, scope)
 237
 238        class_deserializers: list[ClassDeserializer] = list(
 239            itertools.chain(
 240                GLOBAL_CLASS_DESERIALIZER, [class_deserializer] if class_deserializer else []
 241            )
 242        )
 243
 244        # Set some globals for all generated functions
 245        g["cls"] = cls
 246        g["serde_scope"] = scope
 247        g["SerdeError"] = SerdeError
 248        g["UserError"] = UserError
 249        g["raise_unsupported_type"] = raise_unsupported_type
 250        g["typename"] = typename
 251        g["ensure"] = ensure
 252        g["typing"] = typing
 253        g["collections"] = collections
 254        g["Literal"] = Literal
 255        g["from_obj"] = from_obj
 256        g["get_generic_arg"] = get_generic_arg
 257        g["is_instance"] = is_instance
 258        g["TypeCheck"] = TypeCheck
 259        g["disabled"] = disabled
 260        g["coerce_object"] = coerce_object
 261        g["_exists_by_aliases"] = _exists_by_aliases
 262        g["_get_by_aliases"] = _get_by_aliases
 263        g["class_deserializers"] = class_deserializers
 264        g["BeartypeCallHintParamViolation"] = BeartypeCallHintParamViolation
 265        if deserializer:
 266            g["serde_legacy_custom_class_deserializer"] = functools.partial(
 267                serde_legacy_custom_class_deserializer, custom=deserializer
 268            )
 269
 270        # Collect types used in the generated code.
 271        for typ in iter_types(cls):
 272            # When we encounter a dataclass not marked with deserialize, then also generate
 273            # deserialize functions for it.
 274            if is_dataclass_without_de(typ):
 275                # We call deserialize and not wrap to make sure that we will use the default serde
 276                # configuration for generating the deserialization function.
 277                deserialize(typ)
 278
 279            # We don't want to add primitive class e.g "str" into the scope, but primitive
 280            # compatible types such as IntEnum and a subclass of primitives are added,
 281            # so that generated code can use those types.
 282            if is_primitive(typ) and not is_enum(typ) and not is_primitive_subclass(typ):
 283                continue
 284
 285            if is_generic(typ):
 286                g[typename(typ)] = get_origin(typ)
 287            else:
 288                g[typename(typ)] = typ
 289
 290        # render all union functions
 291        for union in iter_unions(cls):
 292            union_args = type_args(union)
 293            add_func(
 294                scope,
 295                union_func_name(UNION_DE_PREFIX, union_args),
 296                render_union_func(cls, union_args, tagging),
 297                g,
 298            )
 299
 300        # render literal functions
 301        for literal in iter_literals(cls):
 302            literal_args = type_args(literal)
 303            add_func(
 304                scope, literal_func_name(literal_args), render_literal_func(cls, literal_args), g
 305            )
 306
 307        # Collect default values and default factories used in the generated code.
 308        for f in defields(cls):
 309            assert f.name
 310            if has_default(f):
 311                scope.defaults[f.name] = f.default
 312            elif has_default_factory(f):
 313                scope.defaults[f.name] = f.default_factory
 314            if f.deserializer:
 315                g[f.deserializer.name] = f.deserializer
 316
 317        add_func(
 318            scope,
 319            FROM_ITER,
 320            render_from_iter(cls, deserializer, type_check, class_deserializer=class_deserializer),
 321            g,
 322        )
 323        add_func(
 324            scope,
 325            FROM_DICT,
 326            render_from_dict(
 327                cls, rename_all, deserializer, type_check, class_deserializer=class_deserializer
 328            ),
 329            g,
 330        )
 331
 332        logger.debug(f"{typename(cls)}: {SERDE_SCOPE} {scope}")
 333
 334        stack.pop()
 335        return cls
 336
 337    if _cls is None:
 338        return wrap  # type: ignore
 339
 340    if _cls in GENERATION_STACK:
 341        return _cls
 342
 343    GENERATION_STACK.append(_cls)
 344    try:
 345        return wrap(_cls)
 346    finally:
 347        GENERATION_STACK.pop()

 350def is_deserializable(instance_or_class: Any) -> bool:
 351    """
 352    Test if an instance or class is deserializable.
 353
 354    >>> @deserialize
 355    ... class Foo:
 356    ...     pass
 357    >>>
 358    >>> is_deserializable(Foo)
 359    True
 360    """
 361    return hasattr(instance_or_class, SERDE_SCOPE)

 364def is_dataclass_without_de(cls: type[Any]) -> bool:
 365    if not dataclasses.is_dataclass(cls):
 366        return False
 367    if not hasattr(cls, SERDE_SCOPE):
 368        return True
 369    scope: Optional[Scope] = getattr(cls, SERDE_SCOPE)
 370    if not scope:
 371        return True
 372    return FROM_DICT not in scope.funcs

 375class Deserializer(Generic[T], metaclass=abc.ABCMeta):
 376    """
 377    `Deserializer` base class. Subclass this to customize deserialize behaviour.
 378
 379    See `serde.json.JsonDeserializer` and `serde.msgpack.MsgPackDeserializer` for example usage.
 380    """
 381
 382    @classmethod
 383    @abc.abstractmethod
 384    def deserialize(cls, data: T, **opts: Any) -> Any:
 385        """
 386        deserialize `data` into an object typically `dict`, `list` or `tuple`.
 387
 388        For example, `serde.json.JsonDeserializer` takes json string and deserialize
 389        into an object. `serde.msgpack.MsgPackDeserializer` takes msgpack bytes and
 390        deserialize into an object.
 391        """
 392        raise NotImplementedError

 395def from_obj(c: type[T], o: Any, named: bool, reuse_instances: Optional[bool]) -> T:
 396    """
 397    Deserialize from an object into an instance of the type specified as arg `c`.
 398    `c` can be either primitive type, `list`, `tuple`, `dict` or `deserialize` class.
 399    """
 400
 401    res: Any
 402
 403    # It is possible that the parser already generates the correct data type requested
 404    # by the caller. Hence, it should be checked early to avoid doing anymore work.
 405    if type(o) is c:
 406        return o
 407
 408    def deserializable_to_obj(cls: type[T]) -> T:
 409        serde_scope: Scope = getattr(cls, SERDE_SCOPE)
 410        func_name = FROM_DICT if named else FROM_ITER
 411        res = serde_scope.funcs[func_name](
 412            cls, maybe_generic=maybe_generic, data=o, reuse_instances=reuse_instances
 413        )
 414        return res  # type: ignore
 415
 416    if is_union(c) and not is_opt(c):
 417        # If a class in the argument is a non-dataclass class e.g. Union[Foo, Bar],
 418        # pyserde generates a wrapper (de)serializable dataclass on the fly,
 419        # and use it to deserialize into the object.
 420        return CACHE.deserialize_union(c, o)
 421
 422    if is_generic(c):
 423        # Store subscripted generic type such as Foo[Bar] in "maybe_generic",
 424        # and store origin type such as Foo in "c". Since subscripted generics
 425        # are not a subclass of "type", use "c" for type inspection, and pass
 426        # "maybe_generic" in deserialize functions.
 427        maybe_generic = c
 428        c = get_origin(c)  # type: ignore
 429    else:
 430        maybe_generic = c
 431    try:
 432        thisfunc = functools.partial(from_obj, named=named, reuse_instances=reuse_instances)
 433        if is_dataclass_without_de(c):
 434            # Do not automatically implement beartype if dataclass without serde decorator
 435            # is passed, because it is surprising for users
 436            # See https://github.com/yukinarit/pyserde/issues/480
 437            deserialize(c, type_check=disabled)
 438            res = deserializable_to_obj(c)
 439        elif is_deserializable(c):
 440            res = deserializable_to_obj(c)
 441        elif is_opt(c):
 442            if o is None:
 443                res = None
 444            else:
 445                res = thisfunc(type_args(c)[0], o)
 446        elif is_list(c):
 447            if is_bare_list(c):
 448                res = list(o)
 449            else:
 450                res = [thisfunc(type_args(c)[0], e) for e in o]
 451        elif is_set(c):
 452            if is_bare_set(c):
 453                res = set(o)
 454            elif is_frozen_set(c):
 455                res = frozenset(thisfunc(type_args(c)[0], e) for e in o)
 456            else:
 457                res = {thisfunc(type_args(c)[0], e) for e in o}
 458        elif is_tuple(c):
 459            if is_bare_tuple(c) or is_variable_tuple(c):
 460                res = tuple(e for e in o)
 461            else:
 462                res = tuple(thisfunc(type_args(c)[i], e) for i, e in enumerate(o))
 463        elif is_dict(c):
 464            if is_bare_dict(c):
 465                res = o
 466            elif is_default_dict(c):
 467                f = DeField(c, "")
 468                v = f.value_field()
 469                origin = get_origin(v.type)
 470                res = collections.defaultdict(
 471                    origin if origin else v.type,
 472                    {
 473                        thisfunc(type_args(c)[0], k): thisfunc(type_args(c)[1], v)
 474                        for k, v in o.items()
 475                    },
 476                )
 477            else:
 478                res = {
 479                    thisfunc(type_args(c)[0], k): thisfunc(type_args(c)[1], v) for k, v in o.items()
 480                }
 481        elif is_numpy_array(c):
 482            res = deserialize_numpy_array_direct(c, o)
 483        elif is_datetime(c):
 484            res = c.fromisoformat(o)
 485        elif is_any(c) or is_ellipsis(c):
 486            res = o
 487        else:
 488            res = c(o)  # type: ignore
 489
 490        return res  # type: ignore
 491
 492    except UserError as e:
 493        raise e.inner from None
 494
 495    except Exception as e:
 496        raise SerdeError(e) from None

 507def from_dict(cls: Any, o: dict[str, Any], reuse_instances: Optional[bool] = None) -> Any:
 508    """
 509    Deserialize dictionary into object.
 510
 511    >>> @deserialize
 512    ... class Foo:
 513    ...     i: int
 514    ...     s: str = 'foo'
 515    ...     f: float = 100.0
 516    ...     b: bool = True
 517    >>>
 518    >>> from_dict(Foo, {'i': 10, 's': 'foo', 'f': 100.0, 'b': True})
 519    Foo(i=10, s='foo', f=100.0, b=True)
 520
 521    You can pass any type supported by pyserde. For example,
 522
 523    >>> lst = [{'i': 10, 's': 'foo', 'f': 100.0, 'b': True},
 524    ...        {'i': 20, 's': 'foo', 'f': 100.0, 'b': True}]
 525    >>> from_dict(list[Foo], lst)
 526    [Foo(i=10, s='foo', f=100.0, b=True), Foo(i=20, s='foo', f=100.0, b=True)]
 527    """
 528    return from_obj(cls, o, named=True, reuse_instances=reuse_instances)

 539def from_tuple(cls: Any, o: Any, reuse_instances: Optional[bool] = None) -> Any:
 540    """
 541    Deserialize tuple into object.
 542

 544    ... class Foo:
 545    ...     i: int
 546    ...     s: str = 'foo'
 547    ...     f: float = 100.0
 548    ...     b: bool = True
 549    >>>
 550    >>> from_tuple(Foo, (10, 'foo', 100.0, True))
 551    Foo(i=10, s='foo', f=100.0, b=True)
 552
 553    You can pass any type supported by pyserde. For example,
 554
 555    >>> lst = [(10, 'foo', 100.0, True), (20, 'foo', 100.0, True)]

 558    """
 559    return from_obj(cls, o, named=False, reuse_instances=reuse_instances)

 562@dataclass
 563class DeField(Field[T]):
 564    """
 565    Represents a field of dataclass.
 566    """
 567
 568    datavar: Optional[str] = None
 569    """ Name of variable which is passed in the deserialize API """
 570
 571    index: int = 0
 572    """ Field index """
 573
 574    iterbased: bool = False
 575    """ Iterater based deserializer or not """
 576
 577    def __getitem__(self, n: int) -> Union[DeField[Any], InnerField[Any]]:
 578        """
 579        Get inner `Field` from current `Field`.
 580
 581        `InnerField` is returned if self is of any standard collection e.g. list.
 582        `DeField` is returned if self is Optional.
 583        """
 584        typ = type_args(self.type)[n]
 585        opts: dict[str, Any] = {
 586            "case": self.case,
 587            "rename": self.rename,
 588            "skip": self.skip,
 589            "skip_if": self.skip_if,
 590            "skip_if_false": self.skip_if_false,
 591            "flatten": self.flatten,
 592            "alias": self.alias,
 593            "parent": self.parent,
 594        }
 595        if is_list(self.type) or is_dict(self.type) or is_set(self.type):
 596            return InnerField(typ, "v", datavar="v", **opts)
 597        elif is_tuple(self.type):
 598            return InnerField(typ, f"{self.data}[{n}]", datavar=f"{self.data}[{n}]", **opts)
 599        else:
 600            # For Optional etc.
 601            return DeField(
 602                typ,
 603                self.name,
 604                datavar=self.datavar,
 605                index=self.index,
 606                iterbased=self.iterbased,
 607                **opts,
 608            )
 609
 610    def key_field(self) -> DeField[Any]:
 611        """
 612        Get inner key field for dict like class.
 613        """
 614        k = self[0]
 615        k.name = "k"
 616        k.datavar = "k"
 617        return k
 618
 619    def value_field(self) -> DeField[Any]:
 620        """
 621        Get inner value field for dict like class.
 622        """
 623        return self[1]
 624
 625    @property
 626    def data(self) -> str:
 627        """
 628        Renders the variable name that possesses the data.
 629
 630        e.g. tuple
 631            * datavar property returns "data"
 632            * data property returns "data[0]".
 633        e.g. Optional
 634            * datavar property returns "data"
 635            * data property returns "data.get("field_name")".
 636        For other types
 637            * datavar property returns "data"
 638            * data property returns "data["field_name"]".
 639        """
 640
 641        if self.iterbased:
 642            return f"{self.datavar}[{self.index}]"
 643        elif is_union(self.type) and type(None) in get_args(self.type):
 644            return f'{self.datavar}.get("{self.conv_name()}")'
 645        else:
 646            return f'{self.datavar}["{self.conv_name()}"]'
 647
 648    @data.setter
 649    def data(self, d: str) -> None:
 650        self.datavar = d
 651
 652    def data_or(self) -> str:
 653        if self.iterbased:
 654            return self.data
 655        else:
 656            return f'{self.datavar}.get("{self.conv_name()}")'

 659@dataclass
 660class InnerField(DeField[T]):
 661    """
 662    Field of Inner type. The purpose of this class is to override "data" method
 663    for inner type codegen.
 664
 665    e.g.
 666      T of list[T]
 667      V of dict[K, V]
 668      T of Optional[T]
 669    """
 670
 671    @property
 672    def data(self) -> str:
 673        return self.datavar or ""
 674
 675    @data.setter
 676    def data(self, d: str) -> None:
 677        self.datavar = d

 690    func: str
 691    cls: Optional[type[Any]] = None
 692    legacy_class_deserializer: Optional[DeserializeFunc] = None
 693    import_numpy: bool = False
 694    suppress_coerce: bool = False
 695    """ Disable type coercing in codegen """
 696    class_deserializer: Optional[ClassDeserializer] = None
 697
 698    def render(self, arg: DeField[Any]) -> str:
 699        """
 700        Render rvalue
 701        """
 702        implemented_methods: dict[type[Any], int] = {}
 703        class_deserializers: Iterable[ClassDeserializer] = itertools.chain(
 704            GLOBAL_CLASS_DESERIALIZER, [self.class_deserializer] if self.class_deserializer else []
 705        )
 706        for n, class_deserializer in enumerate(class_deserializers):
 707            for sig in class_deserializer.__class__.deserialize.methods:  # type: ignore
 708                implemented_methods[get_args(sig.types[1])[0]] = n
 709
 710        custom_deserializer_available = arg.type in implemented_methods
 711        if custom_deserializer_available and not arg.deserializer:
 712            res = (
 713                f"class_deserializers[{implemented_methods[arg.type]}].deserialize("
 714                f"{typename(arg.type)}, {arg.data})"
 715            )
 716        elif arg.deserializer and arg.deserializer.inner is not default_deserializer:
 717            res = self.custom_field_deserializer(arg)
 718        elif is_generic(arg.type):
 719            arg.type_args = list(get_args(arg.type))
 720            origin = get_origin(arg.type)
 721            assert origin
 722            arg.type = origin
 723            res = self.render(arg)
 724        elif is_dataclass(arg.type):
 725            res = self.dataclass(arg)
 726        elif is_opt(arg.type):
 727            res = self.opt(arg)
 728        elif is_list(arg.type):
 729            res = self.list(arg)
 730        elif is_set(arg.type):
 731            res = self.set(arg)
 732        elif is_dict(arg.type):
 733            res = self.dict(arg)
 734        elif is_tuple(arg.type):
 735            res = self.tuple(arg)
 736        elif is_enum(arg.type):
 737            res = self.enum(arg)
 738        elif is_numpy_scalar(arg.type):
 739            self.import_numpy = True
 740            res = deserialize_numpy_scalar(arg)
 741        elif is_numpy_array(arg.type):
 742            self.import_numpy = True
 743            res = deserialize_numpy_array(arg)
 744        elif is_union(arg.type):
 745            res = self.union_func(arg)
 746        elif is_str_serializable(arg.type):
 747            res = f"({self.c_tor_with_check(arg)}) if reuse_instances else {self.c_tor(arg)}"
 748        elif is_datetime(arg.type):
 749            from_iso = f"{typename(arg.type)}.fromisoformat({arg.data})"
 750            res = f"({arg.data} if isinstance({arg.data}, {typename(arg.type)}) else {from_iso}) \
 751                    if reuse_instances else {from_iso}"
 752        elif is_none(arg.type):
 753            res = "None"
 754        elif is_any(arg.type) or is_ellipsis(arg.type):
 755            res = arg.data
 756        elif is_primitive(arg.type):
 757            # For subclasses for primitives e.g. class FooStr(str), coercing is always enabled
 758            res = self.primitive(arg, not is_primitive_subclass(arg.type))
 759        elif isinstance(arg.type, TypeVar):
 760            if not self.cls:
 761                raise SerdeError("Missing cls")
 762            index = find_generic_arg(self.cls, arg.type)
 763            res = (
 764                f"from_obj(get_generic_arg(maybe_generic, maybe_generic_type_vars, "
 765                f"variable_type_args, {index}), {arg.data}, named={not arg.iterbased}, "
 766                "reuse_instances=reuse_instances)"
 767            )
 768        elif is_literal(arg.type):
 769            res = self.literal(arg)
 770        else:
 771            return f"raise_unsupported_type({arg.data})"
 772
 773        if arg.supports_default():
 774            res = self.default(arg, res)
 775
 776        if (
 777            self.legacy_class_deserializer
 778            and not arg.deserializer
 779            and not custom_deserializer_available
 780        ):
 781            # Rerender the code for default deserializer.
 782            default = Renderer(
 783                self.func, self.cls, None, suppress_coerce=self.suppress_coerce
 784            ).render(arg)
 785            return self.custom_class_deserializer(arg, default)
 786        else:
 787            return res
 788
 789    def custom_field_deserializer(self, arg: DeField[Any]) -> str:
 790        """
 791        Render rvalue for the field with custom deserializer.
 792        """
 793        if not arg.deserializer:
 794            raise SerdeError("Missing custom field deserializer")
 795        return f"{arg.deserializer.name}({arg.data})"

 797    def custom_class_deserializer(self, arg: DeField[Any], code: str) -> str:
 798        """
 799        Render custom class deserializer.
 800        """
 801        # The function takes a closure in order to execute the default value lazily.
 802        return (
 803            "serde_legacy_custom_class_deserializer("
 804            f"{typename(arg.type)}, "
 805            f"{arg.datavar}, "
 806            f"{arg.data_or()}, "
 807            f"default=lambda: {code})"
 808        )

 810    def dataclass(self, arg: DeField[Any]) -> str:
 811        if not arg.flatten:
 812            # e.g. "data['field']" will be used as variable name.
 813            var = arg.data
 814        else:
 815            # Because the field is flattened
 816            # e.g. "data" will be used as variable name.
 817            assert arg.datavar
 818            if arg.iterbased:
 819                var = f"{arg.datavar}[{arg.index}:]"
 820            else:
 821                var = arg.datavar
 822
 823        type_args_str = [str(t).lstrip("~") for t in arg.type_args] if arg.type_args else None
 824
 825        opts = (
 826            "maybe_generic=maybe_generic, maybe_generic_type_vars=maybe_generic_type_vars, "
 827            f"variable_type_args={type_args_str}, reuse_instances=reuse_instances"
 828        )
 829
 830        if arg.is_self_referencing():
 831            class_name = "cls"
 832        else:
 833            class_name = typename(arg.type)
 834
 835        return f"{class_name}.{SERDE_SCOPE}.funcs['{self.func}'](data={var}, {opts})"

 837    def opt(self, arg: DeField[Any]) -> str:
 838        """
 839        Render rvalue for Optional.
 840
 841        >>> Renderer('foo').render(DeField(Optional[int], 'o', datavar='data'))
 842        '(coerce_object(int, data["o"])) if data.get("o") is not None else None'
 843
 844        >>> Renderer('foo').render(DeField(Optional[list[int]], 'o', datavar='data'))
 845        '([coerce_object(int, v) for v in data["o"]]) if data.get("o") is not None else None'
 846
 847        >>> Renderer('foo').render(DeField(Optional[list[int]], 'o', datavar='data'))
 848        '([coerce_object(int, v) for v in data["o"]]) if data.get("o") is not None else None'
 849
 850        >>> @deserialize
 851        ... class Foo:
 852        ...     o: Optional[list[int]]
 853        >>> Renderer('foo').render(DeField(Optional[Foo], 'f', datavar='data'))
 854        '(Foo.__serde__.funcs[\\'foo\\'](data=data["f"], maybe_generic=maybe_generic, \
 855maybe_generic_type_vars=maybe_generic_type_vars, variable_type_args=None, \
 856reuse_instances=reuse_instances)) if data.get("f") is not None else None'
 857        """
 858        value_arg = arg[0]
 859        if arg.iterbased:
 860            exists = f"{arg.data} is not None"
 861        else:
 862            name = arg.conv_name()
 863            if arg.alias:
 864                aliases = (f'"{s}"' for s in [name, *arg.alias])
 865                get = f"_get_by_aliases(data, [{','.join(aliases)}], raise_error=False)"
 866            else:
 867                get = f'{arg.datavar}.get("{name}")'
 868            exists = f"{get} is not None"
 869        return f"({self.render(value_arg)}) if {exists} else None"

 871    def list(self, arg: DeField[Any]) -> str:
 872        """
 873        Render rvalue for list.
 874
 875        >>> Renderer('foo').render(DeField(list[int], 'l', datavar='data'))
 876        '[coerce_object(int, v) for v in data["l"]]'

 878        >>> Renderer('foo').render(DeField(list[list[int]], 'l', datavar='data'))
 879        '[[coerce_object(int, v) for v in v] for v in data["l"]]'
 880        """
 881        if is_bare_list(arg.type):
 882            return f"list({arg.data})"
 883        else:
 884            return f"[{self.render(arg[0])} for v in {arg.data}]"

 886    def set(self, arg: DeField[Any]) -> str:
 887        """
 888        Render rvalue for set.
 889
 890        >>> from typing import Set
 891        >>> Renderer('foo').render(DeField(Set[int], 'l', datavar='data'))
 892        'set(coerce_object(int, v) for v in data["l"])'
 893
 894        >>> Renderer('foo').render(DeField(Set[Set[int]], 'l', datavar='data'))
 895        'set(set(coerce_object(int, v) for v in v) for v in data["l"])'
 896        """
 897        if is_bare_set(arg.type):
 898            return f"set({arg.data})"
 899        elif is_frozen_set(arg.type):
 900            return f"frozenset({self.render(arg[0])} for v in {arg.data})"
 901        else:
 902            return f"set({self.render(arg[0])} for v in {arg.data})"

 904    def tuple(self, arg: DeField[Any]) -> str:
 905        """
 906        Render rvalue for tuple.
 907
 908        >>> @deserialize
 909        ... class Foo: pass
 910        >>> Renderer('foo').render(DeField(tuple[str, int, list[int], Foo], 'd', datavar='data'))
 911        '(coerce_object(str, data["d"][0]), coerce_object(int, data["d"][1]), \
 912[coerce_object(int, v) for v in data["d"][2]], \
 913Foo.__serde__.funcs[\\'foo\\'](data=data["d"][3], maybe_generic=maybe_generic, \
 914maybe_generic_type_vars=maybe_generic_type_vars, variable_type_args=None, \
 915reuse_instances=reuse_instances),)'
 916
 917        >>> field = DeField(tuple[str, int, list[int], Foo],
 918        ...                'd',
 919        ...                 datavar='data',
 920        ...                 index=0,
 921        ...                 iterbased=True)
 922        >>> Renderer('foo').render(field)
 923        "(coerce_object(str, data[0][0]), coerce_object(int, data[0][1]), \
 924[coerce_object(int, v) for v in data[0][2]], Foo.__serde__.funcs['foo'](data=data[0][3], \
 925maybe_generic=maybe_generic, maybe_generic_type_vars=maybe_generic_type_vars, \
 926variable_type_args=None, reuse_instances=reuse_instances),)"
 927        """
 928        if is_bare_tuple(arg.type):
 929            return f"tuple({arg.data})"
 930        elif is_variable_tuple(arg.type):
 931            earg = arg[0]
 932            earg.datavar = "v"
 933            return f"tuple({self.render(earg)} for v in {arg.data})"
 934        else:
 935            values = []
 936            for i, _typ in enumerate(type_args(arg.type)):
 937                inner = arg[i]
 938                values.append(self.render(inner))
 939            return f'({", ".join(values)},)'  # trailing , is required for single element tuples
 940
 941    def dict(self, arg: DeField[Any]) -> str:
 942        """
 943        Render rvalue for dict.
 944
 945        >>> Renderer('foo').render(DeField(dict[str, int], 'd', datavar='data'))
 946        '{coerce_object(str, k): coerce_object(int, v) for k, v in data["d"].items()}'
 947
 948        >>> @deserialize
 949        ... class Foo: pass
 950        >>> Renderer('foo').render(DeField(dict[Foo, list[Foo]], 'f', datavar='data'))
 951        '\
 952{Foo.__serde__.funcs[\\'foo\\'](data=k, maybe_generic=maybe_generic, \
 953maybe_generic_type_vars=maybe_generic_type_vars, variable_type_args=None, \
 954reuse_instances=reuse_instances): \
 955[Foo.__serde__.funcs[\\'foo\\'](data=v, maybe_generic=maybe_generic, \
 956maybe_generic_type_vars=maybe_generic_type_vars, \
 957variable_type_args=None, reuse_instances=reuse_instances) for v in v] \
 958for k, v in data["f"].items()}'
 959        """
 960        if is_bare_dict(arg.type):
 961            return arg.data
 962        elif is_default_dict(arg.type):
 963            k = arg.key_field()
 964            v = arg.value_field()
 965            origin = get_origin(v.type)
 966            if origin:
 967                # When the callable type is of generic type e.g list.
 968                # Get origin type "list" from "list[X]".
 969                callable = origin.__name__
 970            else:
 971                # When the callable type is non generic type e.g int, Foo.
 972                callable = v.type.__name__
 973            return f"collections.defaultdict({callable}, \
 974                    {{{self.render(k)}: {self.render(v)} for k, v in {arg.data}.items()}})"
 975        else:
 976            k = arg.key_field()
 977            v = arg.value_field()
 978            return f"{{{self.render(k)}: {self.render(v)} for k, v in {arg.data}.items()}}"
 979
 980    def enum(self, arg: DeField[Any]) -> str:
 981        return f"{typename(arg.type)}({self.primitive(arg)})"

 983    def primitive(self, arg: DeField[Any], suppress_coerce: bool = False) -> str:
 984        """
 985        Render rvalue for primitives.
 986
 987        * `suppress_coerce`: Overrides "suppress_coerce" in the Renderer's field
 988
 989        >>> Renderer('foo').render(DeField(int, 'i', datavar='data'))
 990        'coerce_object(int, data["i"])'
 991
 992        >>> Renderer('foo').render(DeField(int, 'int_field', datavar='data', case='camelcase'))
 993        'coerce_object(int, data["intField"])'
 994
 995        >>> Renderer('foo').render(DeField(int, 'i', datavar='data', index=1, iterbased=True))
 996        'coerce_object(int, data[1])'
 997        """
 998        typ = typename(arg.type)
 999        dat = arg.data
1000        if arg.alias:
1001            aliases = (f'"{s}"' for s in [arg.name, *arg.alias])
1002            dat = f"_get_by_aliases(data, [{','.join(aliases)}])"
1003        if self.suppress_coerce and suppress_coerce:
1004            return dat
1005        else:
1006            return f"coerce_object({typ}, {dat})"

1065def render_from_iter(
1066    cls: type[Any],
1067    legacy_class_deserializer: Optional[DeserializeFunc] = None,
1068    type_check: TypeCheck = strict,
1069    class_deserializer: Optional[ClassDeserializer] = None,
1070) -> str:
1071    template = """
1072def {{func}}(cls=cls, maybe_generic=None, maybe_generic_type_vars=None, data=None,
1073             variable_type_args=None, reuse_instances=None):
1074  if reuse_instances is None:
1075    reuse_instances = {{serde_scope.reuse_instances_default}}
1076
1077  maybe_generic_type_vars = maybe_generic_type_vars or {{cls_type_vars}}
1078
1079  {% for f in fields %}
1080  __{{f.name}} = {{f|arg(loop.index-1)|rvalue}}
1081  {% endfor %}
1082
1083  try:
1084    return cls(
1085      {% for f in fields %}
1086      __{{f.name}},
1087      {% endfor %}
1088    )
1089  except BeartypeCallHintParamViolation as e:
1090    raise SerdeError(e)
1091  except Exception as e:
1092    raise UserError(e)
1093    """
1094
1095    renderer = Renderer(
1096        FROM_ITER,
1097        cls=cls,
1098        legacy_class_deserializer=legacy_class_deserializer,
1099        suppress_coerce=(not type_check.is_coerce()),
1100        class_deserializer=class_deserializer,
1101    )
1102    env = jinja2.Environment(loader=jinja2.DictLoader({"iter": template}))
1103    env.filters.update({"rvalue": renderer.render})
1104    env.filters.update({"arg": to_iter_arg})
1105    fields = list(filter(renderable, defields(cls)))
1106    res = env.get_template("iter").render(
1107        func=FROM_ITER,
1108        serde_scope=getattr(cls, SERDE_SCOPE),
1109        fields=fields,
1110        cls_type_vars=get_type_var_names(cls),
1111    )
1112
1113    if renderer.import_numpy:
1114        res = "import numpy\n" + res
1115
1116    return res

1179def render_union_func(
1180    cls: type[Any], union_args: Sequence[type[Any]], tagging: Tagging = DefaultTagging
1181) -> str:
1182    template = """
1183def {{func}}(cls=cls, maybe_generic=None, maybe_generic_type_vars=None, data=None,
1184             variable_type_args=None, reuse_instances = {{serde_scope.reuse_instances_default}}):
1185  errors = []
1186  {% for t in union_args %}
1187  try:
1188    # create fake dict so we can reuse the normal render function
1189    {% if tagging.is_external() and is_taggable(t)  %}
1190    ensure("{{t|typename}}" in data , "'{{t|typename}}' key is not present")
1191    fake_dict = {"fake_key": data["{{t|typename}}"]}
1192
1193    {% elif tagging.is_internal() and is_taggable(t) %}
1194    ensure("{{tagging.tag}}" in data , "'{{tagging.tag}}' key is not present")
1195    ensure("{{t|typename}}" == data["{{tagging.tag}}"], "tag '{{t|typename}}' isn't found")
1196    fake_dict = {"fake_key": data}
1197
1198    {% elif tagging.is_adjacent() and is_taggable(t) %}
1199    ensure("{{tagging.tag}}" in data , "'{{tagging.tag}}' key is not present")
1200    ensure("{{tagging.content}}" in data , "'{{tagging.content}}' key is not present")
1201    ensure("{{t|typename}}" == data["{{tagging.tag}}"], "tag '{{t|typename}}' isn't found")
1202    fake_dict = {"fake_key": data["{{tagging.content}}"]}
1203
1204    {% else %}
1205    fake_dict = {"fake_key": data}
1206    {% endif %}
1207
1208    {% if t|is_primitive or t|is_none %}
1209    if not isinstance(fake_dict["fake_key"], {{t|typename}}):
1210        raise Exception("Not a type of {{t|typename}}")
1211    {% endif %}
1212    return {{t|arg|rvalue}}
1213  except Exception as e:
1214    errors.append(f' Failed to deserialize into {{t|typename}}: {e}')
1215  {% endfor %}
1216  raise SerdeError("Can not deserialize " + repr(data) + " of type " + \
1217          typename(type(data)) + " into {{union_name}}.\\nReasons:\\n" + "\\n".join(errors))
1218    """

183@dataclass_transform()
184def deserialize(
185    _cls: Optional[type[T]] = None,
186    rename_all: Optional[str] = None,
187    reuse_instances_default: bool = True,
188    convert_sets_default: bool = False,
189    deserializer: Optional[DeserializeFunc] = None,
190    tagging: Tagging = DefaultTagging,
191    type_check: TypeCheck = strict,
192    class_deserializer: Optional[ClassDeserializer] = None,
193    **kwargs: Any,
194) -> type[T]:
195    """
196    A dataclass with this decorator is deserializable from any of the data formats supported
197    by pyserde.
198
199    >>> from serde import deserialize
200    >>> from serde.json import from_json
201    >>>
202    >>> @deserialize
203    ... class Foo:
204    ...     i: int
205    ...     s: str
206    ...     f: float
207    ...     b: bool
208    >>>
209    >>> from_json(Foo, '{"i": 10, "s": "foo", "f": 100.0, "b": true}')
210    Foo(i=10, s='foo', f=100.0, b=True)
211    """
212
213    stack = []
214
215    def wrap(cls: type[T]) -> type[T]:
216        if cls in stack:
217            return cls
218        stack.append(cls)
219
220        tagging.check()
221
222        # If no `dataclass` found in the class, dataclassify it automatically.
223        if not is_dataclass(cls):
224            dataclass(cls)
225
226        if type_check.is_strict():
227            beartype.beartype(cls)
228
229        g: dict[str, Any] = {}
230
231        # Create a scope storage used by serde.
232        # Each class should get own scope. Child classes can not share scope with parent class.
233        # That's why we need the "scope.cls is not cls" check.
234        scope: Optional[Scope] = getattr(cls, SERDE_SCOPE, None)
235        if scope is None or scope.cls is not cls:
236            scope = Scope(cls, reuse_instances_default=reuse_instances_default)
237            setattr(cls, SERDE_SCOPE, scope)
238
239        class_deserializers: list[ClassDeserializer] = list(
240            itertools.chain(
241                GLOBAL_CLASS_DESERIALIZER, [class_deserializer] if class_deserializer else []
242            )
243        )
244
245        # Set some globals for all generated functions
246        g["cls"] = cls
247        g["serde_scope"] = scope
248        g["SerdeError"] = SerdeError
249        g["UserError"] = UserError
250        g["raise_unsupported_type"] = raise_unsupported_type
251        g["typename"] = typename
252        g["ensure"] = ensure
253        g["typing"] = typing
254        g["collections"] = collections
255        g["Literal"] = Literal
256        g["from_obj"] = from_obj
257        g["get_generic_arg"] = get_generic_arg
258        g["is_instance"] = is_instance
259        g["TypeCheck"] = TypeCheck
260        g["disabled"] = disabled
261        g["coerce_object"] = coerce_object
262        g["_exists_by_aliases"] = _exists_by_aliases
263        g["_get_by_aliases"] = _get_by_aliases
264        g["class_deserializers"] = class_deserializers
265        g["BeartypeCallHintParamViolation"] = BeartypeCallHintParamViolation
266        if deserializer:
267            g["serde_legacy_custom_class_deserializer"] = functools.partial(
268                serde_legacy_custom_class_deserializer, custom=deserializer
269            )
270
271        # Collect types used in the generated code.
272        for typ in iter_types(cls):
273            # When we encounter a dataclass not marked with deserialize, then also generate
274            # deserialize functions for it.
275            if is_dataclass_without_de(typ):
276                # We call deserialize and not wrap to make sure that we will use the default serde
277                # configuration for generating the deserialization function.
278                deserialize(typ)
279
280            # We don't want to add primitive class e.g "str" into the scope, but primitive
281            # compatible types such as IntEnum and a subclass of primitives are added,
282            # so that generated code can use those types.
283            if is_primitive(typ) and not is_enum(typ) and not is_primitive_subclass(typ):
284                continue
285
286            if is_generic(typ):
287                g[typename(typ)] = get_origin(typ)
288            else:
289                g[typename(typ)] = typ
290
291        # render all union functions
292        for union in iter_unions(cls):
293            union_args = type_args(union)
294            add_func(
295                scope,
296                union_func_name(UNION_DE_PREFIX, union_args),
297                render_union_func(cls, union_args, tagging),
298                g,
299            )
300
301        # render literal functions
302        for literal in iter_literals(cls):
303            literal_args = type_args(literal)
304            add_func(
305                scope, literal_func_name(literal_args), render_literal_func(cls, literal_args), g
306            )
307
308        # Collect default values and default factories used in the generated code.
309        for f in defields(cls):
310            assert f.name
311            if has_default(f):
312                scope.defaults[f.name] = f.default
313            elif has_default_factory(f):
314                scope.defaults[f.name] = f.default_factory
315            if f.deserializer:
316                g[f.deserializer.name] = f.deserializer
317
318        add_func(
319            scope,
320            FROM_ITER,
321            render_from_iter(cls, deserializer, type_check, class_deserializer=class_deserializer),
322            g,
323        )
324        add_func(
325            scope,
326            FROM_DICT,
327            render_from_dict(
328                cls, rename_all, deserializer, type_check, class_deserializer=class_deserializer
329            ),
330            g,
331        )
332
333        logger.debug(f"{typename(cls)}: {SERDE_SCOPE} {scope}")
334
335        stack.pop()
336        return cls
337
338    if _cls is None:
339        return wrap  # type: ignore
340
341    if _cls in GENERATION_STACK:
342        return _cls
343
344    GENERATION_STACK.append(_cls)
345    try:
346        return wrap(_cls)
347    finally:
348        GENERATION_STACK.pop()
